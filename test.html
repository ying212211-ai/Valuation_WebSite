<script>

function calculate() {

  const copies = +copiesEl.value;

  const pages = +pagesEl.value;

  const priceType = +priceTypeEl.value;


  const printPrice = +document.getElementById(printType.value).value;

  const paperPrice = +document.getElementById(paperType.value).value;

  const bindingPrice = +document.getElementById(bindingType.value).value;

  const processPrice = +document.getElementById(processType.value).value;


  const printingFee = printPrice * pages * copies;

  const paperFee = paperPrice * pages * copies;

  const bindingFee = bindingPrice * copies;

  const processFee = processPrice * copies;


  const subtotal = (printingFee + paperFee + bindingFee + processFee) * priceType;

  const total = subtotal * 1.05;


  result.innerHTML = `

    成品尺寸：${sizeW.value || "-"} × ${sizeH.value || "-"} mm<br>

    紙張：${paperType.options[paperType.selectedIndex].text} ${paperWeight.value} 磅<hr>

    印刷費用：${printingFee}<br>

    紙張費用：${paperFee}<br>

    裝訂費用：${bindingFee}<br>

    後加工費用：${processFee}<hr>

    （以上小計 × 價格類別）× 1.05<br>

    <div class="total">含稅後金額：${Math.round(total)}</div>

  `;

}


const copiesEl = document.getElementById("copies");

const pagesEl = document.getElementById("pages");

const priceTypeEl = document.getElementById("priceType");

const printType = document.getElementById("printType");

const paperType = document.getElementById("paperType");

const paperWeight = document.getElementById("paperWeight");

const bindingType = document.getElementById("bindingType");

const processType = document.getElementById("processType");

const sizeW = document.getElementById("sizeW");

const sizeH = document.getElementById("sizeH");

const result = document.getElementById("result");

</script>
